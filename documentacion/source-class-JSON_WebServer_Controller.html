 11:  * <p>Los métodos ser invocarán pasando la información de la llamada en formato
 12:  * JSON. La información trasmitida en el objeto JSON tendrá la siguiente 
 13:  * estructura array('method'=>'nombre', 'arguments'=>'argumentos')<br/>
 14:  * Donde</p>
 15:  * <ul>
 16:  *   <li>method: Será el nombre del metodo del controlador a invocar, que deberá
 17:  *               estar definido como public o protected</li>
 18:  *   <li>arguments: Serán los argumentos a pasar a la función. Será un valor
 19:  *      simple o un array si hay más de un parámetro.</li>
 20:  * </ul>
 21:  * <p>Las funciones que se implementen en el controlador deberán ser public o 
 22:  * protected para que puedan ser invocadas desde la clase 
 23:  * JSON_WebServer_Controller</p>
 24:  * 
 25:  * <p>Para cada método llamado se devolverá en JSON un objeto con el siguiente
 26:  * formato array('error'=>'', 'return'=>'', 'debug'=>'')</p>
 27:  * <ul>
 28:  *  <li>error: Indica si la llamada ha tenido (FALSE) o no (TRUE) exito</li>
 29:  *  <li>return: valor devuelto. Si hay error será la descripción del error
 30:  *      en otro caso lo que devuelva la función.</li>
 31:  *  <li>debug: Si está activado el modo depuración en el servidor
 32:  *      se devolverá información sobre los parámetros recibidos</li>
 33:  * </ul>
 34:  * <p>Ejemplo de uso</p>
 35:  * <pre>
 36:         require_once(APPPATH.'/libraries/JSON_WebServer_Controller.php');

113:      *  - error: Indica si la llamada ha tenido o no exito
114:      *  - return: valor devuelto. Si hay error será la descripción del error

121:     public function Index()
122:     {
123:         $method =$_SERVER['REQUEST_METHOD'];
124:         
125:         // Dependiendo del método de la petición ejecutaremos la acción correspondiente.
126:         switch ($method) {
127:             case 'POST':
128:                 // Solo atendemos llamadas cuando nos llegan por POST
129:                 $this->ProcessCall();
130:                 break;
131:             case 'GET':
132:             default:

139:     /**
140:      * Procesa la llamada realizada desde un cliente utilizando POST. 
141:      * En el cuerpo del mensaje se espera recibir el nombre del método a 
142:      * invocar y los parámetros que recibirá.
143:      */
144:     private function ProcessCall()
145:     {
146:         // Obtenemos los parametros de entrada
147:         $input_params=$this->ProcesaParametrosJSON();
148: 
149:         //
150:         // Realizamos comprobaciones con método y parámetros
151:         // 
152:         // Comprobamos si han indicado el método
153:         if ( ! isset($input_params['method']) )
154:         {
155:             $this->ReturnError(
156:                'No se ha indicado el método que se debe invocar. Envíe en ' .
157:                'el POST, JSON con el formato ("method":XXX, "arguments":XXX)');
158:         }
159:         $method_name=$input_params['method'];
160:         if (! method_exists($this, $method_name))
161:         {
162:             $this->ReturnError(
163:                'No está declarado el método <'.$method_name.'> en el servicio');
164:         }
165: 
166:         // Han indicado los argumentos de la función
167:         if ( ! isset($input_params['arguments']) )
168:         {
169:             $this->ReturnError(
170:                'No se ha indicado los argumentos del método. Envíe en ' .
171:                'el POST, JSON con el formato ("method":XXX, "arguments":XXX)');
172:         }
173:         else if (! is_array($input_params['arguments']))
174:         {
175:             // Los argumentos no son un array, pero deseamos que lo sea
176:             $aguments=array($input_params['arguments']);
177:         }
178:         else
179:         {   // Los argumentos son un array
180:             $arguments=$input_params['arguments'];
181:         }
182: 
183:         //
184:         // Llamamos al metodo seleccionado
185:         // Evitamos mostrar cualquier mensaje de error que se produzca. Si
186:         // no se indican los parámetros apropiados se sustituirán por NULL
187:         $return_value=
188:                 call_user_func_array(array($this, $method_name), $arguments);
189: 
190:         if ($return_value===NULL)
191:         {
192:             // Ha habido error en la llamada
193:             $str_args='';
194:             foreach($arguments as $arg)
195:             {
196:                 $str_args.=($str_args!==''?', ':'').$arg;
197:             }
198:             $this->ReturnError(
199:                     "Error en la llamada al metodo [$method_name] ".
200:                     "con los parámetros [$str_args]");
201:         }
202: 
203:         // Todo correcto devolvemos el valor
204:         $this->ReturnValues($return_value);
205:     }

235:      * 
236:      * Los parámetros se enviarán en el cuerpo del mensaje
237:      */
238:     private function ProcesaParametrosJSON()
239:     {
240:         /* Los parametros los recibiremos en formato JSON mediante POST
241:          * Dichos parametros se recogerán de la variable 
242:          * $HTTP_RAW_POST_DATA que contiene los datos de los parametros 
243:          * sin procesar
244:          * http://www.php.net/manual/en/reserved.variables.httprawpostdata.php
245:          * 
246:          * No se puede utilizar $_POST pues este array procesa los datos
247:          * recibidos y los transforma en un array. 
248:          * El problema es que no vienen en el formato apropiado, vienen
249:          * en formato JSON
250:          *  
251:          *  Más información:
252:          *  http://stackoverflow.com/questions/12194751/what-is-the-raw-post-data
253:          *  http://stackoverflow.com/questions/3173547/whats-the-difference-between-post-and-raw-post-in-php-at-all
254:          *      
255:          **/
256: 
257:         //
258:         // Obtenemos la información que viene en el POST sin procesar
259:         // No podemos utilizar la variable $_POST pues está la genera 
260:         // automáticamente PHP pero cuando los datos vienen en otro formato
261:         //
262:         $HTTP_RAW_POST_DATA = file_get_contents('php://input');
263: 
264:         if ($this->debug)
265:         {
266:             $this->debug_input_params=$HTTP_RAW_POST_DATA;
267:         }
268: 
269:         //
270:         // Comprobamos que hayan enviado parámetros
271:         //
272:         if (! isset($HTTP_RAW_POST_DATA))
273:         {
274:             // Error fin del script
275:             $this->RetornaError('No se han enviado datos JSON para operar');
276: 
277:             /*
278:              * Otra forma de obtener los datos enviado con POST podría ser 
279:              * (http://php.net/manual/es/wrappers.php.php)
280:              * 
281:              * $rawPost=file_get_contents('php://input');
282:              */
283:         }       
284:         else 
285:         {
286:             // Devolvemos los parametros como array
287:             return json_decode($HTTP_RAW_POST_DATA, TRUE);
288:         }
289:     }
